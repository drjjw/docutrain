# Quick troubleshooting commands - run these on the remote server (already SSH'd in)

# 1. Check PM2 status and recent logs (MOST IMPORTANT)
pm2 status && echo "=== LAST 50 LINES OF LOGS ===" && pm2 logs docutrainio-bot --lines 50 --nostream

# 2. Check if .env file exists (CRITICAL - deploy script excludes it!)
ls -la /home/docutrainio/public_html/.env || echo "❌ .env file MISSING - this is likely the problem!"

# 3. Check if server responds
curl -s http://localhost:3458/api/health || echo "❌ Server not responding"

# 4. Check PM2 detailed info
pm2 info docutrainio-bot

# 5. If .env is missing, you'll need to copy it from your local machine (run this locally):
# scp -i ~/.ssh/drjjw.pub -P 7022 .env root@162.246.254.111:/home/docutrainio/public_html/.env

# 6. After copying .env, restart:
cd /home/docutrainio/public_html && pm2 restart docutrainio-bot && sleep 3 && pm2 logs docutrainio-bot --lines 30 --nostream
